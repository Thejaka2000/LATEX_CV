\ProjectsSection

\Project
{Supersonic Disk--Gap--Band Parachute CFD}
{TAFSM}
{2025 -- Present}
{
Developed numerical simulations of compressible flow around
supersonic Disk--Gap--Band parachutes for Martian atmospheric
entry using Isogeometric Analysis.
}

\vspace{0.010cm}

\Project
{Shock Wave Oscillation Analysis}
{TAFSM}
{2025}
{
Investigated oscillatory shock-wave behaviour beneath
Disk--Gap--Band parachutes and analysed vortex dynamics,
pressure fluctuations and transient flow structures.
}

\vspace{0.010cm}

\Project
{Fluid--Structure Interaction of DGB Parachutes}
{TAFSM}
{2025}
{
Developed computational models coupling structural deformation
with compressible fluid flow using Isogeometric Analysis.
}
