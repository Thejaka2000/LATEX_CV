%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Resume Start / End
%% \beginresume / \endresume were referenced but never defined anywhere in
%% the project, which is why \ResumeStart threw "Undefined control sequence."
%% These now open/close resume.cls's `cvcolumns` environment directly.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand{\ResumeStart}{%
  \begin{cvcolumns}
  \cvleftcolumn
}

\newcommand{\ResumeEnd}{%
  \end{cvcolumns}
  \resumeend
}


%% Summary
\newcommand{\Summary}[1]{%
  \cvsection{Summary}
  {\BodyFont #1\par}%
}

%% Experience Entry
% #1 Job Title  #2 Company  #3 Date  #4 Location  #5 \item bullet points
\newcommand{\Experience}[5]{%
  \experienceentry
    {#1}
    {#2}
    {#3}
    {#4}
    {\begin{resumelist}#5\end{resumelist}}
}

%% Education Entry
% #1 Degree  #2 University  #3 Year  #4 Location
\newcommand{\Education}[4]{%
  \educationentry{#1}{#2}{#3}{#4}
}


%% Project Entry
% #1 Project  #2 Organization  #3 Date  #4 Description
\newcommand{\Project}[4]{%
  \projectentry{#1}{#2}{#3}{#4}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Two Column Helpers
%% Delegate to resume.cls's own \cvleftcolumn/\cvrightcolumn instead of
%% calling \switchcolumn directly here. The `paracol` package reserves the
%% literal names \leftcolumn/\rightcolumn/\switchcolumn for its own internal
%% (and mutually-poisoning) shorthand -- resume.cls already works around
%% that once, so this file just reuses that fix rather than duplicating it.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\LeftColumn}{\cvleftcolumn}
\newcommand{\RightColumn}{\cvrightcolumn}
